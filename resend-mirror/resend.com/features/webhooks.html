Resend sends a POST to your route with a JSON payload of details.

import type {
  NextRequest
} from "next/server";

export const POST = async (
  req: NextRequest
) => {
  const event = await req.json();

  if (event.type === 'email.delivered') {
    return NextResponse.json(event);
  }

  return NextResponse.json({});
};
